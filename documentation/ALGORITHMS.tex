 \begin{algorithm}[H]
 \caption{Algorithm to update the content for InfoWindow Object on bus stop marker}
 \begin{algorithmic}[1]
 \renewcommand{\algorithmicrequire}{\textbf{Input:}}
 \renewcommand{\algorithmicensure}{\textbf{Output:}}
 \REQUIRE \textit{'route' and 'bus\_stop\_id'}
 \ENSURE {The estimated and scheduled arrival times are updated}
  \\ \textit{Initialisation} :
  \FOR{ every 5 seconds}
            \IF{An InfoWindow is Open}
                % \STATE{Invoke the \textit{callBackMethod()}}
                \STATE Call \textit{getEstimatedArrivalAJAX} with input parameters.
                  \IF{data}
                        \IF{data['est\_arrival'] is not empty}
                            \IF{est\_arrival is equal to defaultTimeString}
                                \STATE{scheduled\_arrival $\leftarrow$ defaultTimeString}
                            \ENDIF
                            \STATE{est\_arrival $\leftarrow$ data['est\_arrival']}
                        \ELSE
                             \STATE{est\_arrival $\leftarrow$ defaultTimeString}
                        \ENDIF
                        \IF{data['scheduled\_arrival'] is not empty}
                                \STATE{scheduled\_arrival $\leftarrow$ data['scheduled\_arrival']}
                        \ENDIF
                    \ELSE
                        \STATE{\textit{scheduled\_arrival} $\leftarrow$ \textit{defaultTimeString}}
                        \STATE{\textit{est\_arrival} $\leftarrow$  \textit{defaultTimeString}}
                    \ENDIF
                    \STATE{Invoke method to Update the Active Info Window Content with latest Details}
            \ENDIF
        \ENDFOR
 \RETURN $None$
 \end{algorithmic}
 \end{algorithm}
